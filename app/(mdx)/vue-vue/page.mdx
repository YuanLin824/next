# Vue

## Options API (vue2)

```html
<!-- 不便于后期维护和复用 -->
<script>
  export default {
    data() {
      return {
        name: "world",
      }
    },
    methods: {
      greet(name) {
        console.log(`Hello, ${name}!`)
      },
    },
    computed: {
      greeting() {
        return `Hello, ${this.name}`
      },
    },
    watch: {
      name(newName, oldName) {
        console.log(`${oldName} -> ${newName}`)
      },
    },

    // vue3 新增的 setup 配置项 (Composition API 的基础)
    // setup 在 beforeCreate 之前执行 (setup 是最早的生命周期)
    // setup 中没有 this (this 为 undefined)
    // setup 将数据和方法通过 return 返回才能在 template 中使用
    // setup 返回值也可以是一个渲染函数
    // data 中能够通过 this.xxx 获取到 setup 中定义的数据 (反之不行)
    setup() {
      const greet = (name) => console.log(`Hello, ${name}!`)
      return { greet }
    },
  }
</script>
```

## Composition API (vue3)

```html
<!-- 用函数的方式, 更加优雅的组织代码, 让相关功能的代码更加有序的组织在一起 -->
<!-- setup 是 vue3 新增的配置项 (Composition API 的基础), 能以语法糖形式使用 -->
<script setup>
  import { ref, reactive } from "vue"

  // ref 一般用来定义基本类型的响应式数据
  // 返回一个 RefImpl 对象 (ref 对象)
  const count = ref(0)
  function increment() {
    // template 中直接使用
    // script 中使用 ref 包装的数据, 需要使用 .value 访问
    count.value++
  }
  // ref 也可以定义复杂类型数据
  const person = ref({ name: "zs", age: 18 })
  function changeAhe() {
    person.value.age++
  }

  // reactive 只能定义复杂类型的响应式数据 (深层响应式)
  // 返回一个 Proxy 对象
  // 直接使用即可
  const car = reactive({ price: 10000, name: "BYD" })
  function changeName() {
    car.name = "Tesla"
  }
</script>
```

## toRef 和 toRefs

```html
<script setup lang="ts">
  const person = reactive({ name: "ZS", age: 18 })

  const name = toRef(person, "name")
  const age = toRef(person, "age")
  // 与上面写法效果一致
  const { name, age } = toRefs(person)

  // toRef 的返回值是 ObjectRefImpl 类型
  // toRefs 的返回值是 key 为属性名, value 为 toRef(person, key) 的对象
  name.value = "ZS_LS"
  age.value = 19
</script>

<template>
  <span>{{ name }}</span>
  <span>{{ age }}</span>

  <span>{{ person.name }}</span>
  <span>{{ person.age }}</span>
</template>
```

## 计算属性

```html
<script setup lang="ts">
  const person = reactive({ name: "ZS", age: 18 })
  // 计算属性 ComputedRefImpl
  // 只读写法
  const fullName = computed(() => `${person.name} -- ${person.age}`)
  // 可读可写写法
  const fullName = computed({
    get() {
      return `${person.name} -- ${person.age}`
    },
    set(val) {
      console.log("set", val)
      const [name, age] = val.split(" -- ")
      person.name = name
      person.age = Number(age)
    },
  })
</script>

<template>
  姓名: <input type="text" v-model:value="person.name" /> <br />
  年龄: <input type="text" v-model:value="person.age" /> <br />
```

## watch

```html
<script setup>
  // 只能监视一下四种数据
  //   1. ref 定义的数据
  //   2. reactive 定义的数据
  //   3. 一个函数, 返回一个值 (即 gutter 函数)
  //   4. 包含上面内容的数组
  import { ref, watch } from "vue"
  const count = ref(0)
  // 监视 ref 定义的基本类型数据 (监视的是 ref 返回的整个 RefImpl 类型的对象)
  const stopWatch1 = watch(count, (newCount, oldCount) => {
    console.log("count changed: ", newCount, oldCount)
    if (true) stopWatch1()
  })

  // 监视 ref 定义的复杂类型数据 (默认监视的是对象的地址值; 可手动开启深度监视)
  const person = ref({ name: "ZS", age: 19 })
  const stopWatch2 = watch(
    person,
    (newVal, oldVal) => {
      console.log("count changed: ", newVal, oldVal)
      if (true) stopWatch2()
    },
    // deep: true; 开启深层监视
    // immediate 开启立即监视; 立即执行一边监视回调 (该次执行时 oldVal 为 undefined)
    { deep: true, immediate: true }
  )

  // 监视 reactive 定义的数据
  const car = reactive({ brand: "BMW", model: "M3" })
  // 监视 reactive 定义的数据时, 默认开启深度监视, 且不可关闭
  const stopWatch3 = watch(car, (newVal, oldVal) => {
    console.log("car changed: ", newVal, oldVal)
    if (true) stopWatch2()
  })

  // 监视 ref 或 reactive 定义的对象的某个基本类型属性
  const animal = reactive({ type: "cat", name: "Mimi", info: { age: 1, color: "white" } })
  const stopWatch4 = watch(
    () => animal.type,
    (newVal, oldVal) => {
      console.log("animal type changed: ", newVal, oldVal)
    }
  )

  // 监视 ref 或 reactive 定义的对象的某个复杂类型属性
  const stopWatch4 = watch(
    () => animal.info,
    (newVal, oldVal) => {
      console.log("animal info changed: ", newVal, oldVal)
    },
    { deep: true }
  )

  // 监视多个数据源
  const stopWatch4 = watch(
    [() => animal.type, () => animal.name, () => animal.info],
    (newVal, oldVal) => {
      console.log("animal type or name changed: ", newVal, oldVal)
    },
    { deep: true }
  )
</script>

<template>
  <button @click="count++">{{ count }}</button>
</template>
```

## 对生命周期的理解

```html
<script setup lang="ts">
  // 父: 创建 -> 子: 创建 -> 子: 挂载 -> 父: 挂载 -> 子: 卸载 -> 父: 卸载

  // 创建: vue2: beforeCreate, created; vue3: setup() {} 就是创建阶段
  // 挂在: vue2: beforeMount, mounted; vue3: onBeforeMount(callback), onMounted(callback)
  // 更新: vue2: beforeUpdate, updated; vue3: onBeforeUpdate(callback), onUpdated(callback)
  // 销毁(vue2)/卸载(vue3): vue2: beforeDestroy, destroyed; vue3: onBeforeUnmount(callback), onUnmounted(callback)
</script>
```

## 自定义 hook

```ts
// useSum.ts
import { ref, onBeforeUnmount } from "vue"
export default function useSum() {
  const sum = ref(0)
  const callbacks = []

  function add(value) {
    sum.value += value
    callbacks.forEach((callback) => callback())
  }

  function remove(value) {
    sum.value -= value
    callbacks.forEach((callback) => callback())
  }

  onBeforeUnmount(() => {
    sum.value = 0
  })

  return { sum, add, remove }
}
```

```html
  import SlotChild from "./SlotChild.vue"

<template>
  <SlotChild>
    <div>默认插槽: 外部传递的内容</div>

    <!-- v-slot 只能用在 template 或组件上-->
    <!-- v-slot 简写为 ## -->
    <template v-slot:nameSlot>
      <div>具名插槽: 外部传递的内容</div>
    </template>

    <!-- /** 👇 作用域插槽 👇 */ -->
    <template v-slot="{ childName }"> childName: {{ childName }} </template>
    <template v-slot:scopeSlot="{ childName }"> childName: {{ childName }} </template>
  </SlotChild>
</template>
```

## 其他 API

### shallowRef 和 shallowReactive

定义浅层响应式对象, 可作为性能优化

### readonly 和 shallowReadonly

- readonly: 将响应式对象变为只读的
- shallowReadonly: 将响应式对象的浅层变为只读的，但是对象内的嵌套对象是可写的

### toRaw 和 markRaw

- toRaw: 将响应式对象变为非响应式对象
- markRaw: 标记一个对象, 使其不会成为响应式对象

## customRef

```ts
import { customRef } from "vue"

export const useMsgRef = (initMsg = "init") => {
  const msg = customRef((track, trigger) => {
    return {
      get() {
        console.log("get called")
        track() // 告诉 vue 对 msg 持续关注 (即跟踪数据变化)
        return initMsg
      },
      set(value) {
        console.log("set called", value)
        initMsg = value
        trigger() // 通知 vue 对 msg 的值有改变
      },
    }
  })

  return { msg }
}
```
