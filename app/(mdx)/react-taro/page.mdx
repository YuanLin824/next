# Taro

## 简介

Taro 是京东开发的开放式跨端跨框架解决方案，支持使用 React / Vue / Preact 等语法来构建微信小程序、支付宝小程序、字节跳动小程序、H5、RN 等应用。

- **一套代码，多端运行** — 统一开发，编译到不同平台
- **类 React 语法** — 使用 JSX、组件化、状态管理
- **丰富的生态** — Taro UI、社区插件、Taroify 等组件库

## 环境搭建

```bash
# 安装 Taro CLI
npm install -g @tarojs/cli

# 创建项目
taro init myApp

# 选择模板: React + TypeScript
# 选择样式方案: Sass / Less / Stylus / CSS Modules
# 选择模板: 默认模板 / MobX / Redux / Vuex

# 进入项目
cd myApp

# 编译运行（微信小程序）
npm run dev:weapp

# 编译运行（H5）
npm run dev:h5

# 生产构建
npm run build:weapp
npm run build:h5
```

## 生命周期

### 页面生命周期

```tsx
import {
  useDidShow,
  useDidHide,
  usePullDownRefresh,
  useReachBottom,
  usePageScroll,
  useShareAppMessage,
} from "@tarojs/taro"

function Page() {
  // 对应 onLoad — 页面加载时触发
  // 对应 onReady — 页面初次渲染完成时触发

  // 页面显示 / 隐藏 — 对应 onShow / onHide
  useDidShow(() => console.log("页面显示"))
  useDidHide(() => console.log("页面隐藏"))

  // 下拉刷新 — 对应 onPullDownRefresh
  usePullDownRefresh(() => {
    console.log("下拉刷新")
    Taro.stopPullDownRefresh() // 停止下拉刷新动画
  })

  // 上拉触底 — 对应 onReachBottom
  useReachBottom(() => console.log("触底加载更多"))

  // 页面滚动
  usePageScroll((res) => console.log(res.scrollTop))

  // 分享 — 对应 onShareAppMessage
  useShareAppMessage(() => ({
    title: "分享标题",
    path: "/pages/index/index",
    imageUrl: "https://example.com/share.png",
  }))

  return <View>Page Content</View>
}

export default Page
```

## 路由跳转与传参

```tsx
import Taro from "@tarojs/taro"
import { Navigator } from "@tarojs/components"

// 声明式导航 — 使用 Navigator 组件
function Page() {
  return (
    <>
      {/* navigateTo — 保留当前页，跳转到新页面 */}
      <Navigator url="/pages/detail/index?id=123">跳转到详情</Navigator>

      {/* redirectTo — 关闭当前页，跳转到新页面 */}
      <Navigator url="/pages/login/index" openType="redirect">
        重定向
      </Navigator>

      {/* switchTab — 跳转到 tabBar 页面 */}
      <Navigator url="/pages/index/index" openType="switchTab">
        回首页
      </Navigator>

      {/* navigateBack — 返回上一页 */}
      <Navigator openType="navigateBack">返回</Navigator>

      {/* reLaunch — 关闭所有页面，打开新页面 */}
      <Navigator url="/pages/index/index" openType="reLaunch">
        重启
      </Navigator>
    </>
  )
}

// 编程式导航
async function navigateToDetail() {
  // 保留当前页面，跳转
  await Taro.navigateTo({ url: "/pages/detail/index?id=123" })
}

function navigateBack() {
  Taro.navigateBack({ delta: 1 })
}

function switchToHome() {
  Taro.switchTab({ url: "/pages/index/index" })
}

function redirectToLogin() {
  Taro.redirectTo({ url: "/pages/login/index" })
}

function reLaunch() {
  Taro.reLaunch({ url: "/pages/index/index" })
}
```

### 路由传参

```tsx
// 发送方：通过 query string 传参
Taro.navigateTo({ url: "/pages/detail/index?id=123&name=taro" })

// 接收方：在目标页面的 useLoad / useDidShow 中获取参数
import { useLoad } from "@tarojs/taro"

function Detail() {
  useLoad((options) => {
    console.log(options.id) // "123"
    console.log(options.name) // "taro"

  // 也可以从路由历史事件中获取
  const instance = Taro.getCurrentInstance()

// 复杂数据传参：通过全局状态管理或全局变量

Taro.setStorageSync("detailData", { id: 1, items: [1, 2, 3] })
```

## 事件处理

```tsx
import { View, Input, Button } from "@tarojs/components"
import { useState } from "react"

function EventDemo() {
  const [value, setValue] = useState("")

  const handleTap = (e: any) => {
    e.stopPropagation() // 阻止冒泡
    console.log("tapped")
  }

  // 注意：小程序中事件回调不支持直接绑定参数，需通过箭头函数
  const handleClick = (id: number) => () => {
    console.log("clicked:", id)
  }

  return (
    <View>
      {/* 点击事件 onClick / onLongPress */}
      <View onClick={handleTap} onLongPress={() => console.log("长按")}>
        点击区域
      </View>

      {/* 输入事件 */}
      <Input value={value} onInput={(e) => setValue(e.detail.value)} />

      {/* 带参数的事件 */}
      <Button onClick={handleClick(123)}>点击传参</Button>

      {/* 表单提交 */}
      <form onSubmit={(e) => console.log(e.detail.value)}>
        <Input name="username" />
        <Button formType="submit">提交</Button>
      </form>
    </View>
  )
}
```

## 条件渲染和列表渲染

```tsx
import { View, Text, Block } from "@tarojs/components"

function ListDemo() {
  const items = [
    { id: 1, name: "项目一" },
    { id: 2, name: "项目二" },
    { id: 3, name: "项目三" },
  ]
  const isLoggedIn = true

  return (
    <View>
      {/* 条件渲染 — 与 React 一致 */}
      {isLoggedIn ? <Text>已登录</Text> : <Text>请登录</Text>}
      {isLoggedIn && <Text>欢迎回来</Text>}

      {/* 列表渲染 — 必须设置 key */}
      {items.map((item) => (
        <View key={item.id} className="item">
          <Text>{item.name}</Text>
        </View>
      ))}

      {/* Block 标签 — 不会渲染为实际 DOM，仅用于逻辑包裹 */}
      <Block>
        {items.map((item) => (
          <View key={item.id}>
            <Text>{item.name}</Text>
          </View>
        ))}
      </Block>
    </View>
  )
}
```

## 数据请求

```tsx
import Taro from "@tarojs/taro"
import { useState, useEffect } from "react"

// Taro.request 是对各平台网络请求的封装
async function fetchData() {
  const res = await Taro.request({
    url: "https://api.example.com/data",
    method: "GET",
    data: { page: 1, size: 10 },
    header: { "Content-Type": "application/json" },
  })
  return res.data
}

// 封装请求工具 — 添加拦截器、Token 注入、错误处理
const request = (options: Taro.request.Option): Promise<any> => {
  const token = Taro.getStorageSync("token")

  return new Promise((resolve, reject) => {
    Taro.request({
      ...options,
      url: `${BASE_URL}${options.url}`,
      header: {
        "Content-Type": "application/json",
        ...(token ? { Authorization: `Bearer ${token}` } : {}),
        ...options.header,
      },
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data)
        } else if (res.statusCode === 401) {
          Taro.showToast({ title: "登录过期", icon: "none" })
          Taro.reLaunch({ url: "/pages/login/index" })
          reject(res)
        } else {
          Taro.showToast({ title: "请求失败", icon: "none" })
          reject(res)
        }
      },
      fail: (err) => {
        Taro.showToast({ title: "网络异常", icon: "none" })
        reject(err)
      },
    })
  })
}

// 在组件中使用
function UserList() {
  const [list, setList] = useState([])

  useEffect(() => {
    request({ url: "/api/users" }).then((data) => setList(data.list))
  }, [])

  return (
    <View>
      {list.map((user: any) => (
        <View key={user.id} className="user-item">
          <Text>{user.name}</Text>
        </View>
      ))}
    </View>
  )
}
```

## 自定义 Hooks

```tsx
import { useState, useEffect } from "react"
import Taro from "@tarojs/taro"

// 封装下拉刷新
function useRefresh(fetchFn: () => Promise<void>) {
  const [refreshing, setRefreshing] = useState(false)

  const onRefresh = async () => {
    setRefreshing(true)
    await fetchFn()
    setRefreshing(false)
  }

  return { refreshing, onRefresh }
}

// 封装页面参数获取
function usePageParams<T>() {
  const [params, setParams] = useState<T>({} as T)

  useEffect(() => {
    const instance = Taro.getCurrentInstance()
    setParams((instance.router?.params ?? {}) as T)
  }, [])

  return params
}

// 封装倒计时
function useCountdown(seconds: number) {
  const [count, setCount] = useState(seconds)
  const [running, setRunning] = useState(false)

  useEffect(() => {
    if (!running || count <= 0) return
    const timer = setInterval(() => setCount((c) => c - 1), 1000)
    return () => clearInterval(timer)
  }, [running, count])

  const start = () => setRunning(true)
  const reset = () => {
    setRunning(false)
    setCount(seconds)
  }

  return { count, running, start, reset }
}

// 使用
function Page() {
  const params = usePageParams<{ id: string }>()
  const { count, start } = useCountdown(60)

  return (
    <View>
      <Text>ID: {params.id}</Text>
      <Text>倒计时: {count}s</Text>
      <Button onClick={start}>开始</Button>
    </View>
  )
}
```

## 跨端兼容注意事项

| 差异点            | 说明                                                                     |
| ----------------- | ------------------------------------------------------------------------ |
| **CSS 支持**      | 小程序不支持 CSS 动画中的某些属性；不支持 fixed 定位在 scroll-view 内    |
| **DOM API**       | 小程序没有 DOM，不能用 `document.querySelector` / `window` / `navigator` |
| **组件名大写**    | 自定义组件在 `app.config.ts` 中注册后才可使用（使用 `usingComponents`）  |
| **HTML 标签**     | 小程序中不能直接用 `<div>` `<span>`，必须用 `<View>` `<Text>` 等         |
| **React DOM API** | `ReactDOM.render`、`createPortal` 在小程序中不可用                       |
| **长列表**        | 小程序中使用 `ScrollView`，建议开启 `enable-back-to-top`                 |
| **包体积限制**    | 微信小程序主包 ≤ 2MB，总包 ≤ 20MB，注意分包策略                          |
