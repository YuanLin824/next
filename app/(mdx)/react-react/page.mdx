### useReducer

```tsx

function reducer(state: number, action: Action): number {

      <button onClick={() => dispatch({ type: "decrement" })}>-</button>

    </>
  )
}
```

### 自定义 Hook

```ts
// 复用有状态逻辑，命名以 use 开头
import { useState, useEffect } from "react"

// 封装数据获取逻辑
function useFetch<T>(url: string) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<Error | null>(null)

  useEffect(() => {
    let cancelled = false
    setLoading(true)

    fetch(url)
      .then((res) => res.json())
      .then((json) => {
        if (!cancelled) setData(json)
      })
      .catch((err) => {
        if (!cancelled) setError(err)
      })
      .finally(() => {
        if (!cancelled) setLoading(false)
      })

    return () => {
      cancelled = true
    }
  }, [url])

  return { data, loading, error }
}

// 封装防抖
function useDebounce<T>(value: T, delay: number): T {
  const [debounced, setDebounced] = useState(value)
  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])
  return debounced
}
```

```tsx
// 通过 Props 传递一个函数来共享渲染逻辑
function MouseTracker({ render }: { render: (pos: { x: number; y: number }) => React.ReactNode }) {
  const [pos, setPos] = useState({ x: 0, y: 0 })

  useEffect(() => {
    const handler = (e: MouseEvent) => setPos({ x: e.clientX, y: e.clientY })
    window.addEventListener("mousemove", handler)
    return () => window.removeEventListener("mousemove", handler)
  }, [])

  return <>{render(pos)}</>
}

function App() {
  return (
    <MouseTracker
      render={({ x, y }) => (
        <span>
          鼠标位置: {x}, {y}
        </span>
      )}
    />
  )
}
```

```tsx
// 函数组件中通过 Hooks 替代，理解其阶段概念即可
class LifecycleDemo extends React.Component {
  // 挂载阶段
  constructor(props) {
    super(props)
  } // 初始化 state / 绑定方法
  static getDerivedStateFromProps(props, state) // 派生 state
  render() // 渲染

  // 错误处理
  static getDerivedStateFromError(error) // 渲染备用 UI
  componentDidCatch(error, info) // 记录错误日志
```

## Suspense 与 Error Boundary

```tsx
import { Suspense } from "react"
import { ErrorBoundary } from "react-error-boundary" // 或自行封装

// Suspense：声明式管理异步加载状态
function App() {
  return (
    <ErrorBoundary fallback={<div>出错了</div>}>
      <Suspense fallback={<Loading />}>
        <AsyncComponent />
      </Suspense>
    </ErrorBoundary>
  )
}

// 类组件 Error Boundary（必须用类组件实现）
class ErrorBoundary extends React.Component<
  { children: React.ReactNode; fallback: React.ReactNode },
  { hasError: boolean }
> {
  constructor(props) {
    super(props)
    this.state = { hasError: false }
  }
  static getDerivedStateFromError() {
    return { hasError: true }
  }
  componentDidCatch(error: Error, info: React.ErrorInfo) {
    console.error(error, info)
  }
  render() {
    return this.state.hasError ? this.props.fallback : this.props.children
  }
}
```

```tsx
import { useActionState, useFormStatus } from "react"

async function submitForm(prevState: unknown, formData: FormData) {
  const name = formData.get("name") as string
  // 调用 API...
  return { success: true, name }
}

function Form() {
  const [state, formAction] = useActionState(submitForm, null)

  return (
    <form action={formAction}>
      <input name="name" />
      <SubmitButton />
      {state?.success && <p>提交成功: {state.name}</p>}
    </form>
  )
}

function SubmitButton() {
  const { pending } = useFormStatus()
  return <button disabled={pending}>{pending ? "提交中..." : "提交"}</button>
}
```

### useOptimistic

```tsx
// 乐观更新 — 立即更新 UI，失败时回滚
import { useOptimistic } from "react"

function TodoList({ todos }: { todos: Todo[] }) {
  const [optimisticTodos, addOptimistic] = useOptimistic(todos, (state, newTodo: Todo) => [
    ...state,
    newTodo,
  ])

  async function handleAdd(formData: FormData) {
    const title = formData.get("title") as string
    addOptimistic({ id: Date.now(), title, completed: false })
    await addTodo(title) // 实际 API 调用
  }
  // ...
}
```

### use

```tsx
import { use } from "react"

// use() 可以在渲染期间读取 Promise / Context
// 可以在条件和循环中使用（与 Hooks 不同）

function UserProfile({ userPromise }: { userPromise: Promise<User> }) {
  const user = use(userPromise) // Suspense 会捕获 pending 状态
  return <div>{user.name}</div>
}

// 也可以读取 Context
const theme = use(ThemeContext)
```

### 其他改进

- **`ref` 作为 prop** — 无需 `forwardRef`，直接从 props 中获取 `ref`
- **`<Context>` 作为 Provider** — 无需 `<Context.Provider>`，直接 `<Context value={...}>...</Context>`
- **Document Metadata** — 在组件中直接渲染 `<title>`, `<meta>`, `<link>` 标签
- **更好的错误报告** — 错误信息中包含组件名和错误来源
